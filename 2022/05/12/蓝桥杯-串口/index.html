1、Transmit

1.HAL_UART_Transmit

打开串口

波特率应设置一致

/*****main.c*****/
uint8_t temp_txbuffer[30]={"test test test"};
int main(void)
{
  uint32_t temp_cnt=0;
  while (1)
  { 
    if(HAL_GetTick()-temp_cnt>1000)
    {
        HAL_UART_Transmit(&huart1,temp_txbuffer,14,10); 
        temp_cnt=HAL_GetTick();
    }
  }
}

HAL_StatusTypeDef HAL_UART_Transmit(UART_HandleTypeDef *huart, uint8_t *pData, uint16_t Size, uint32_t Timeout)

1/波特率=发送一个bit需要的时间，一个字符10bits

需要Timeout（超时时间）超过发送内容所需的时间t=1/波特率*10 *Size（字符长度）才能完整发送

2.HAL_UART_Transmit_IT
/*****main.c*****/
uint8_t temp_txbuffer[30]={"test test test"};
int main(void)
{
  uint32_t temp_cnt=0;
  while (1)
  { 
    if(HAL_GetTick()-temp_cnt>1000)
    {
        HAL_UART_Transmit_IT(&huart1,temp_txbuffer,14);//不需要超时时间
        temp_cnt=HAL_GetTick();
    }
  }
}

波特率应不用设置一致，不必等待超时时间

static void UART_RxISR_8BIT(UART_HandleTypeDef *huart)
{
  uint16_t uhMask = huart->Mask;
  uint16_t uhdata;

	if(__HAL_UART_GET_FLAG(huart,USART_ISR_RXNE_RXFNE)!=RESET)   //RX非空 接收数据
	{
		uhdata = (uint16_t) READ_REG(huart->Instance->RDR);
    *huart->pRxBuffPtr = (uint8_t)(uhdata & (uint8_t)uhMask); 
    huart->pRxBuffPtr++; //huart->pRxBuffPtr为地址 地址++ 即往后填
    huart->RxXferCount++;//改成计数++
		
		
	}
	else			//RX空

		huart->RxXferSize=huart->RxXferCount; //用huart->RxXferSize来记录上一次接收字符的帧值
