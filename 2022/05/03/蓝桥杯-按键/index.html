3、GPIO
//按键
PA0 GPIO_INPUT
PB0 GPIO_INPUT
PB1 GPIO_INPUT
PB2 GPIO_INPUT
//LED
PC8 GPIO_OUTPUT
PC9 GPIO_OUTPUT
PC10 GPIO_OUTPUT
PC11 GPIO_OUTPUT

PB0-PB2：

GPIO mode: Input mode

GPIO Pull-up/Pull-down: No pull-up and no pull-down

User Label：KEY1-KEY3（标签）

PD8-PD15:

GPIO output level: High（默认状态）——LED默认关闭

GPIO mode: Output Push Pull（推挽输出：可接高低电平）

Maximum Output speed: High（切换电平状态速度）

User Label：LED1-LED8（标签）

PD2：

GPIO output level: High（默认状态）——LED使能打开

GPIO mode: Output Push Pull（推挽输出：可接高低电平）

Maximum Output speed: High（切换电平状态速度）

User Label: LED1_EN（标签）

Imput frequency：24——HSE——PLLCLK倍频——HCLK：170（回车）

1、消抖&只做一次

/*****key.c*****/
#include "key.h"

uint16_t key_delay_filter_count[4]={0,0,0,0};//数组
key_status keys_status[4];
//↑类型（自定义）  ↑变量
    
void Update_Key_Status(uint8_t key_num)//消抖10ms
{
	if(HAL_GPIO_ReadPin(KEY1_GPIO_Port,KEY1_Pin)=0)
	{
		key_delay_filter_count[key_num]<<1;//左移一位 00000011-->00000110(有16位 这里省略了8位0)
		key_delay_filter_count[key_num]|=0x0001;//与1取或 -->00000111
		if(key_delay_filter_count[key_num]>0x03FF)//3是两个1 FF是8个1 共10位 即10ms延时
		{
            if(keys_status[key_num]=key_released)
                keys_status[key_num]=key_pressed;	
		}
	}
	else
	{
		key_delay_filter_count[key_num]>>1;
		if(key_delay_filter_count[key_num]=0)
		{
			keys_status=key_released;
		}
	}
}

uint16_t Get_Key_Status(uint8_t key_num)
{
    return keys_status[key_num]
}

void Do_It_Once(uint8_t key_num)
{
    keys_status[key_num]=key_waiting_for_release;
}
